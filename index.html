Example 2: Multi Bar Plot

HTML:

TypeScript:


            interface PlotData {
                categories: string[];
                series1: number[];
                series2: number[];
              }
              
              interface PlotSettings {
                title: string;
                xLabel: string;
                yLabel: string;
                barColors: string[];
                legendLabels: string[];
              }
              
              function range(start: number, end: number, step: number = 1): number[] {
                return Array.from({ length: (end - start) / step + 1 }, (_, i) => start + i * step);
              }
              
              document.addEventListener("DOMContentLoaded", () => {
                const data: PlotData = {
                  categories: ['A', 'B', 'C', 'D', 'E'],
                  series1: [5, 3, 4, 7, 2],
                  series2: [3, 2, 5, 6, 4]
                };
              
                const settings: PlotSettings = {
                  title: "Comparison of Series 1 and Series 2",
                  xLabel: "Categories",
                  yLabel: "Values",
                  barColors: ['#FF0000', '#0000FF'],
                  legendLabels: ['Series 1', 'Series 2']
                };
              
                const example2 = document.getElementById("example2") as any; // Assuming easy-canvas can be interacted with like this.
                example2.drawingLoop = function() {
                  this.hotAndReady.multiBarPlot(data, settings);
                }.bind(example2);
              });
